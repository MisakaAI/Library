# 自定义 dialect

## 创建 XML 文件

```xml
<mavlink>
  <include>common.xml</include>
  <messages>
    <message id="900" name="MAG_SENSOR">
      <description>Magnetic sensor data</description>
      <field type="uint64_t" name="time_usec">Timestamp</field>
      <field type="uint8_t" name="sensor_id">Sensor ID</field>
      <field type="float" name="mag">Magnetic Field (nT)</field>
    </message>
    <message id="901" name="FLUXGATE_SENSOR">
      <description>Fluxgate sensor data</description>
      <field type="uint64_t" name="time_usec">Timestamp</field>
      <field type="float" name="x">Mag X (nT)</field>
      <field type="float" name="y">Mag Y (nT)</field>
      <field type="float" name="z">Mag Z (nT)</field>
    </message>
  </messages>
</mavlink>
```

## 生成 MAVLink 库

参考 [安装 MAVLink 工具链](0.Install_MAVLink.md) 进行安装

```sh
cd mavlink

```

### 使用 Python

```sh
python3 -m pymavlink.tools.mavgen --lang=Python3 --wire-protocol=2.0 --output=my_sensor.py message_definitions/v1.0/my_sensor.xml
```

**1. 发送端**

```python
# send.py
from pymavlink import mavutil
import my_sensor
import time
import random

# 创建UDP连接（发送到本地14550）
master = mavutil.mavlink_connection("udpout:127.0.0.1:14550")
master.mav = my_sensor.MAVLink(master)

while True:
    mag_value = round(random.uniform(10000, 99999.99), 2)

    msg = master.mav.mag_sensor_encode(
        int(time.time() * 1e6),  # time_usec
        1,  # 传感器编号
        mag_value,  # 随机磁场值
    )
    master.mav.send(msg)
    print(f"msgId: {msg.get_msgId()} Sent {msg.get_type()} Value {mag_value}")
    time.sleep(1)
```

**2. 接收端**

```python
# receive.py
from pymavlink import mavutil

# 在创建连接时通过 dialect 参数直接指定你的模块名
# 确保 my_sensor.py 在当前目录下或 Python 路径中
connection = mavutil.mavlink_connection("udpin:0.0.0.0:14550", dialect="my_sensor")

print("正在等待 MAVLink 数据 (Dialect: my_sensor)...")

while True:
    try:
        # 阻塞读取
        msg = connection.recv_match(blocking=True)

        if not msg:
            continue

        # 此时 msg 不再是 UNKNOWN，而是能够识别的对象
        if msg.get_type() == "MAG_SENSOR":
            # 直接通过属性访问字段
            print(f"ID: {msg.sensor_id} | Mag: {msg.mag:.2f} | Time: {msg.time_usec}")

        elif msg.get_type() == "BAD_DATA":
            # 如果依然出现 BAD_DATA，通常是 MAVLink 1.0 vs 2.0 协议版本不匹配
            if mavutil.all_printable(msg.data):
                print(f"收到异常数据: {msg.data}")

    except KeyboardInterrupt:
        break
```

### 使用 C

```sh
python3 -m pymavlink.tools.mavgen --lang=C --wire-protocol=2.0 --output=generated/include/mavlink/v2.0 message_definitions/v1.0/my_sensor.xml
```

**1. 发送端**

```c
// send.c
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <unistd.h>
#include <time.h>
#include <arpa/inet.h>

#include <mavlink/v2.0/my_sensor/mavlink.h>

int main()
{
    int sock;
    struct sockaddr_in addr;

    // 创建 UDP socket
    sock = socket(AF_INET, SOCK_DGRAM, 0);

    addr.sin_family = AF_INET;
    addr.sin_port = htons(14550);
    addr.sin_addr.s_addr = inet_addr("127.0.0.1");

    uint8_t system_id = 1;
    uint8_t component_id = 1;

    srand(time(NULL));

    while (1)
    {
        mavlink_message_t msg;
        uint8_t buffer[MAVLINK_MAX_PACKET_LEN];

        uint64_t time_usec = (uint64_t)time(NULL) * 1000000ULL;
        uint8_t sensor_id = 1;
        float mag = 10000 + (rand() % 90000);

        // 打包消息
        mavlink_msg_mag_sensor_pack(system_id, component_id, &msg,
                                    time_usec,
                                    sensor_id,
                                    mag);

        // 转为字节流
        uint16_t len = mavlink_msg_to_send_buffer(buffer, &msg);

        // 发送
        sendto(sock, buffer, len, 0,
               (struct sockaddr *)&addr, sizeof(addr));

        printf("Sent MAG_SENSOR: %.2f\n", mag);

        sleep(1);
    }

    close(sock);
    return 0;
}
```

```sh
# 编译 发送端
gcc send.c -Igenerated/include -o send -Wno-address-of-packed-member
```

**2. 接收端**

```c
// recv.c
#include <stdio.h>
#include <stdint.h>
#include <unistd.h>
#include <arpa/inet.h>

#include <mavlink/v2.0/my_sensor/mavlink.h>

int main()
{
    int sock;
    struct sockaddr_in addr;

    sock = socket(AF_INET, SOCK_DGRAM, 0);

    addr.sin_family = AF_INET;
    addr.sin_port = htons(14550);
    addr.sin_addr.s_addr = INADDR_ANY;

    bind(sock, (struct sockaddr *)&addr, sizeof(addr));

    printf("Listening on UDP 14550...\n");

    mavlink_message_t msg;
    mavlink_status_t status;

    uint8_t buffer[2048];

    while (1)
    {
        int len = recvfrom(sock, buffer, sizeof(buffer), 0, NULL, NULL);

        for (int i = 0; i < len; i++)
        {
            if (mavlink_parse_char(MAVLINK_COMM_0, buffer[i], &msg, &status))
            {

                if (msg.msgid == MAVLINK_MSG_ID_MAG_SENSOR)
                {
                    mavlink_mag_sensor_t data;
                    mavlink_msg_mag_sensor_decode(&msg, &data);

                    printf("MAG_SENSOR:\n");
                    printf("  time: %llu\n", data.time_usec);
                    printf("  sensor_id: %d\n", data.sensor_id);
                    printf("  mag: %.2f\n\n", data.mag);
                }
            }
        }
    }

    close(sock);
    return 0;
}
```

```sh
# 编译 接收端
gcc recv.c -Igenerated/include -o recv -Wno-address-of-packed-member
```
